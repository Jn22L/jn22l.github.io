##### ES6 Arrow Function 테스트
간결하고 멋진듯

```javascript
var a = [
  "Hydrogen",
  "Helium",
  "Lithium",
  "한글"
];

var a2 = a.map(function(s){ return s.length });

console.log(a2); // logs [8, 6, 7, 2]

var a3 = a.map( s => s.length );

console.log(a3); // logs [8, 6, 7, 2]
```
